🔴 KRITIKAL #1 — Data Loss: save_selected_groups
========================================

**FILE:** bot/database.py (line ~484)

**MASALAH:**
DELETE semua kumpulan → INSERT baru. Jika INSERT gagal, semua kumpulan user HILANG selama-lamanya.

**FIX YANG DIPERLUKAN:**
```python
async def save_selected_groups(user_id: int, group_ids: list[int]) -> bool:
    """
    Simpan senarai kumpulan terpilih dengan backup untuk elak data loss.
    """
    try:
        # 1. BACKUP kumpulan lama dulu
        backup_query = """
        SELECT group_id FROM selected_groups 
        WHERE user_id = %s
        """
        backup_result = supabase.rpc('execute_sql', {
            'query': backup_query,
            'params': [user_id]
        }).execute()
        
        old_groups = [row['group_id'] for row in backup_result.data] if backup_result.data else []
        
        # 2. DELETE kumpulan lama
        supabase.table('selected_groups').delete().eq('user_id', user_id).execute()
        
        # 3. INSERT kumpulan baru
        if group_ids:
            rows = [{'user_id': user_id, 'group_id': gid} for gid in group_ids]
            insert_result = supabase.table('selected_groups').insert(rows).execute()
            
            # 4. VERIFY insert berjaya
            if not insert_result.data:
                # INSERT GAGAL! Pulihkan backup
                if old_groups:
                    restore_rows = [{'user_id': user_id, 'group_id': gid} for gid in old_groups]
                    supabase.table('selected_groups').insert(restore_rows).execute()
                    logging.error(f"[save_selected_groups] INSERT gagal, backup dipulihkan untuk user {user_id}")
                return False
        
        return True
        
    except Exception as e:
        logging.error(f"[save_selected_groups] Error: {e}")
        # Jika error, cuba pulihkan backup
        if old_groups:
            try:
                restore_rows = [{'user_id': user_id, 'group_id': gid} for gid in old_groups]
                supabase.table('selected_groups').insert(restore_rows).execute()
                logging.info(f"[save_selected_groups] Backup dipulihkan untuk user {user_id}")
            except:
                pass
        return False
```

🔴 KRITIKAL #2 — Data Loss: transfer_userbot_session
========================================

**FILE:** bot/database.py (line ~446)

**MASALAH:**
DELETE session lama → upsert ke user baru. Jika upsert gagal, session hilang selamanya.

**FIX YANG DIPERLUKAN:**
```python
async def transfer_userbot_session(from_user_id: int, to_user_id: int, userbot_id: str) -> bool:
    """
    Pindah session userbot dari satu user ke user lain.
    ATOMIC: upsert dulu, baru delete. Jika upsert gagal, session lama kekal selamat.
    """
    try:
        # 1. Dapatkan session lama
        old_session = supabase.table('sessions')\
            .select('*')\
            .eq('user_id', from_user_id)\
            .eq('userbot_id', userbot_id)\
            .single()\
            .execute()
        
        if not old_session.data:
            logging.error(f"[transfer_userbot_session] Session tidak dijumpai: {userbot_id}")
            return False
        
        session_data = old_session.data
        
        # 2. UPSERT ke user baru DULU (sebelum delete lama)
        new_session = {
            'user_id': to_user_id,
            'userbot_id': userbot_id,
            'session_string': session_data['session_string'],
            'tg_username': session_data.get('tg_username'),
            'connected_at': session_data.get('connected_at'),
            'is_active': session_data.get('is_active', True),
            'backup_email': session_data.get('backup_email'),
        }
        
        upsert_result = supabase.table('sessions').upsert(new_session).execute()
        
        # 3. VERIFY upsert berjaya
        if not upsert_result.data:
            logging.error(f"[transfer_userbot_session] Upsert gagal untuk user {to_user_id}")
            raise Exception("Upsert session gagal")
        
        # 4. Baru DELETE session lama (selepas verify upsert OK)
        supabase.table('sessions')\
            .delete()\
            .eq('user_id', from_user_id)\
            .eq('userbot_id', userbot_id)\
            .execute()
        
        logging.info(f"[transfer_userbot_session] Berjaya pindah {userbot_id} dari {from_user_id} ke {to_user_id}")
        return True
        
    except Exception as e:
        logging.error(f"[transfer_userbot_session] Error: {e}")
        # Session lama kekal selamat kerana DELETE tidak dipanggil jika upsert gagal
        return False
```

⚠️ AMARAN #2 — transfer_coins() Not Atomic
========================================

**FILE:** bot/database.py (line ~144)

**MASALAH:**
3 DB calls berasingan (deduct → add → log), race condition possible.

**FIX YANG DIPERLUKAN:**

Buat Supabase RPC function untuk atomic transfer:

**1. Cipta function dalam Supabase SQL Editor:**

```sql
CREATE OR REPLACE FUNCTION transfer_coins(
    p_from_user_id INT,
    p_to_user_id INT,
    p_amount INT,
    p_description TEXT DEFAULT 'Transfer'
)
RETURNS JSONB
LANGUAGE plpgsql
AS $$
DECLARE
    v_from_balance INT;
BEGIN
    -- 1. Lock dan check balance sender
    SELECT balance INTO v_from_balance
    FROM wallets
    WHERE user_id = p_from_user_id
    FOR UPDATE;
    
    IF v_from_balance IS NULL THEN
        RETURN jsonb_build_object('success', false, 'error', 'Sender wallet not found');
    END IF;
    
    IF v_from_balance < p_amount THEN
        RETURN jsonb_build_object('success', false, 'error', 'Insufficient balance');
    END IF;
    
    -- 2. Deduct dari sender
    UPDATE wallets
    SET balance = balance - p_amount
    WHERE user_id = p_from_user_id;
    
    -- 3. Add ke receiver
    UPDATE wallets
    SET balance = balance + p_amount
    WHERE user_id = p_to_user_id;
    
    -- 4. Log transaction
    INSERT INTO transactions (user_id, amount, transaction_type, description)
    VALUES 
        (p_from_user_id, -p_amount, 'transfer_out', p_description),
        (p_to_user_id, p_amount, 'transfer_in', p_description);
    
    RETURN jsonb_build_object('success', true);
    
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object('success', false, 'error', SQLERRM);
END;
$$;
```

**2. Update function dalam database.py:**

```python
async def create_subscription(user_id: int, plan_name: str, duration_days: int) -> bool:
    """
    Cipta subscription dengan expires_at calculation.
    """
    try:
        # 1. Matikan subscription lama
```

```python
import asyncio
import logging
from datetime import datetime
from database import supabase

async def check_expired_subscriptions():
    """
    Semak dan matikan subscription yang tamat tempoh.
    Jalankan setiap 1 jam.
    """
    while True:
        try:
            now = datetime.utcnow().isoformat()
            
            # Cari subscription yang tamat dan masih aktif
            expired = supabase.table('subscriptions')\
                .select('id, user_id, plan_name')\
                .eq('active', True)\
                .lt('expires_at', now)\
                .execute()
            
            if expired.data:
                for sub in expired.data:
                    # Matikan subscription
                    supabase.table('subscriptions')\
                        .update({'active': False})\
                        .eq('id', sub['id'])\
                        .execute()
                    
                    logging.info(f"[check_expired_subscriptions] Matikan subscription {sub['plan_name']} untuk user {sub['user_id']}")
            
        except Exception as e:
            logging.error(f"[check_expired_subscriptions] Error: {e}")
        
        # Sleep 1 jam
        await asyncio.sleep(3600)
```
