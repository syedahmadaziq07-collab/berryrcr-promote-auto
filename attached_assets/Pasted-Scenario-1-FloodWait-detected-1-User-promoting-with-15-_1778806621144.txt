Scenario 1: FloodWait detected
1. User promoting with 15 min delay
2. Telegram returns FloodWait(2700s)
3. Bot activates safe mode:
   - Original: 15 min
   - Safe: 54 min (2700s × 1.2 / 60)
4. User gets notification
5. Promote continues with 54 min delay
6. After 2 hours cooldown
7. Bot auto-restores to 15 min
8. User gets restore notification

Scenario 2: PeerFlood detected
1. User promoting with 20 min delay
2. Telegram returns PeerFlood
3. Bot activates safe mode (high risk):
   - Original: 20 min
   - Safe: 100 min (20 × 5)
   - Pause: 1 hour
4. User gets notification
5. Promote pauses 1 hour, then continues with 100 min delay
6. After 2 hours cooldown
7. Auto-restore to 20 min

CREATE TABLE IF NOT EXISTS safe_mode_status (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    userbot_id VARCHAR(50),
    safe_mode_active BOOLEAN DEFAULT FALSE,
    original_delay INTEGER NOT NULL,
    safe_delay INTEGER NOT NULL,
    reason TEXT,
    risk_level VARCHAR(20),
    activated_at TIMESTAMPTZ DEFAULT NOW(),
    cooldown_until TIMESTAMPTZ,
    restored_at TIMESTAMPTZ,
    UNIQUE(user_id, userbot_id)
);

CREATE INDEX IF NOT EXISTS idx_safe_mode_active ON safe_mode_status(safe_mode_active);
CREATE INDEX IF NOT EXISTS idx_safe_mode_cooldown ON safe_mode_status(cooldown_until);
