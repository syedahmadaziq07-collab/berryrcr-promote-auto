4. Fix process_phone — pastikan FSM state 
   disimpan dalam _pending:

   @router.message(UserbotStates.waiting_phone)
   async def process_phone(message: Message,
                            state: FSMContext):
       try:
           # Ambil nombor dari contact atau teks
           if message.contact:
               phone = message.contact.phone_number
               if not phone.startswith("+"):
                   phone = "+" + phone
           else:
               phone = message.text.strip()
               if not phone.startswith("+"):
                   phone = "+" + phone
           
           uid = message.from_user.id
           
           # Jangan hantar OTP jika dah ada 
           # dalam _pending
           if uid in _pending:
               await _cleanup_pending(uid)
           
           await message.answer("⏳ Menghantar kod OTP...")
           
           client = TelegramClient(
               StringSession(),
               API_ID,
               API_HASH
           )
           await client.connect()
           
           result = await client.send_code_request(phone)
           
           # Simpan dalam _pending
           _pending[uid] = {
               "client": client,
               "phone": phone,
               "phone_code_hash": result.phone_code_hash
           }
           
           # Set FSM state
           await state.set_state(UserbotStates.waiting_otp)
           
           await message.answer(
               "📱 Kod OTP telah dihantar ke nombor anda.\n\n"
               "Sila taip kod OTP anda.\n"
               "Contoh: 12345\n\n"
               "⚠️ Jangan kongsi kod ini dengan sesiapa.",
               reply_markup=cancel_keyboard()
           )
           
       except Exception as e:
           print(f"process_phone error: {e}")
           await _cleanup_pending(message.from_user.id)
           await state.clear()
           await message.answer(
               "❌ Gagal menghantar kod OTP.\n"
               "Sila semak nombor telefon dan cuba lagi."
           )
