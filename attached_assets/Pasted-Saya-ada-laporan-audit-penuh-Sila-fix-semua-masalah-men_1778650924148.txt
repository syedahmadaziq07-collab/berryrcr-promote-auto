FIX K2-01 & K2-02 — Race condition wallet
Tukar add_coins dan deduct_coins supaya guna 
operasi atom terus dalam Supabase:

# GANTI kod lama baca-tulis dengan RPC Supabase:
# Untuk add_coins:
await client.rpc('add_coins', {
    'p_user_id': user_id, 
    'p_amount': amount
}).execute()

# Untuk deduct_coins:
await client.rpc('deduct_coins', {
    'p_user_id': user_id,
    'p_amount': amount
}).execute()

# Buat fungsi RPC dalam Supabase SQL Editor:
CREATE OR REPLACE FUNCTION add_coins(
    p_user_id BIGINT, 
    p_amount INTEGER
)
RETURNS void AS $$
BEGIN
    INSERT INTO wallets (user_id, coins, updated_at)
    VALUES (p_user_id, p_amount, NOW())
    ON CONFLICT (user_id) 
    DO UPDATE SET 
        coins = wallets.coins + p_amount,
        updated_at = NOW();
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION deduct_coins(
    p_user_id BIGINT,
    p_amount INTEGER
)
RETURNS void AS $$
BEGIN
    UPDATE wallets 
    SET coins = coins - p_amount,
        updated_at = NOW()
    WHERE user_id = p_user_id
    AND coins >= p_amount;
    
    IF NOT FOUND THEN
        RAISE EXCEPTION 'Baki tidak mencukupi';
    END IF;
END;
$$ LANGUAGE plpgsql;
