Helper function for checking active status:
   def is_schedule_active(start: str, end: str) -> bool:
       """
       Check if current time is within schedule.
       Handles overnight ranges (e.g. 20:00 - 02:00).
       
       Args:
           start: "HH:MM"
           end: "HH:MM"
       
       Returns:
           True if bot should be active now
       """
       from datetime import datetime
       import pytz
       
       tz = pytz.timezone('Asia/Kuala_Lumpur')
       now = datetime.now(tz)
       current = now.strftime("%H:%M")
       
       if end < start:
           # Overnight (crosses midnight)
           # Active if: current >= start OR current <= end
           return current >= start or current <= end
       else:
           # Same day
           # Active if: start <= current <= end
           return start <= current <= end
